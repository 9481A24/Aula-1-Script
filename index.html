from scapy.all import ARP, Ether, srp

def scan(rede):
    arp = ARP(pdst=rede)
    ether = Ether(dst="ff:ff:ff:ff:ff:ff")
    pacote = ether/arp

    result = srp(pacote, timeout=3, verbose=0)[0]

    dispositivos = []
    for enviado, recebido in result:
        dispositivos.append({"ip": recebido.psrc, "mac": recebido.hwsrc})

    return dispositivos

rede = "192.168.0.1/24"

print("Dispositivos encontrados na rede:")
print("IP\t\tMAC")
for dispositivo in dispositivos:
    print(f"{dispositivo['ip']}\t{dispositivo['mac']}")
